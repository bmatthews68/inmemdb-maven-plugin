17  package com.btmatthews.maven.plugins.inmemdb;
18  
19  import java.text.MessageFormat;
20  import java.util.ResourceBundle;
21  
22  /**
23   * Helpers used to load localized message strings.
24   *
25   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
26   * @since 1.2.0
27   */
28  public final class MessageUtil {
29  
30      /**
31       * The base name of the message resource bundle.
32       */
33      private static final String BUNDLE_NAME = "com.btmatthews.maven.plugins.inmemdb.messages";
34  
35      /**
36       * The message resource bundle.
37       */
38      private static final ResourceBundle RESOURCE_BUNDLE = ResourceBundle.getBundle(BUNDLE_NAME);
39  
40      /**
41       * Make the constructor private because this class only offers static utility methods.
42       */
43      private MessageUtil() {
44      }
45  
46      /**
47       * Retrieve the message pattern for a given message key from the resource
48       * bundle and format the message replacing the place holders with the
49       * message arguments.
50       *
51       * @param messageKey The key of the message in the resource bundle.
52       * @param arguments  The message arguments.
53       * @return The expanded message string.
54       */
55      public static String getMessage(final String messageKey,
56                                      final Object... arguments) {
57          final String messagePattern = RESOURCE_BUNDLE.getString(messageKey);
58          return MessageFormat.format(messagePattern, arguments);
59      }
60  }
