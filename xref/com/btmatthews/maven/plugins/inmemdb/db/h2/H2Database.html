37  /**
38   * Implements support for in-memory H2 databases.
39   *
40   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
41   * @version 1.0.0
42   */
43  public final class H2Database extends AbstractSQLDatabase {
44  
45      /**
46       * The connection protocol for in-memory H2 databases.
47       */
48      private static final String PROTOCOL = "h2:tcp://localhost:{0,number,#}/mem:";
49      /**
50       * Default port H2 listens on, can be altered via setting port property
51       */
52      private static final int DEFAULT_PORT = 9092;
53      /**
54       * The loaders that are supported for loading data or executing scripts.
55       */
56      private static final Loader[] LOADERS = new Loader[]{
57              new DBUnitXMLLoader(), new DBUnitFlatXMLLoader(),
58              new DBUnitCSVLoader(), new DBUnitXLSLoader(), new SQLLoader()};
59      /**
60       * The H2 TCP server.
61       */
62      private TcpServer service;
63  
64      /**
65       * The default constructor initializes the default database port.
66       */
67      public H2Database() {
68          super(DEFAULT_PORT);
69      }
70  
71      /**
72       * Get the database connection protocol.
73       *
74       * @return Always returns {@link H2Database#PROTOCOL}.
75       */
76      protected String getUrlProtocol() {
77          return MessageFormat.format(PROTOCOL, getPort());
78      }
79  
80      /**
81       * Get the data source that describes the connection to the in-memory H2
82       * database.
83       *
84       * @return Returns {@code dataSource} which was initialised by the
85       *         constructor.
86       */
87      @Override
88      public DataSource getDataSource() {
89          final Map<String, String> attributes = new HashMap<String, String>();
90          attributes.put("DB_CLOSE_DELAY", "-1");
91          final JdbcDataSource dataSource = new JdbcDataSource();
92          dataSource.setURL(getUrl(attributes));
93          dataSource.setUser(getUsername());
94          dataSource.setPassword(getPassword());
95          return dataSource;
96      }
97  
98      /**
99       * Get the loaders that are supported for loading data or executing scripts.
100      *
101      * @return Returns {@link #LOADERS}.
102      */
103     @Override
104     public Loader[] getLoaders() {
105         return LOADERS;
106     }
107 
108     /**
109      * Start the in-memory H2 database.
110      *
111      * @param logger Used to report errors and raise exceptions.
112      */
113     @Override
114     public void start(final Logger logger) {
115 
116         logger.logInfo("Starting embedded H2 database");
117 
118         try {
119             service = new TcpServer();
120             service.init("-tcpDaemon", "-tcpPort", Integer.toString(getPort()));
121             service.start();
122         } catch (final SQLException exception) {
123             final String message = MessageUtil.getMessage(ERROR_STARTING_SERVER, getDatabaseName());
124             logger.logError(message, exception);
125             return;
126         }
127 
128         final Thread serviceListenerThread = new Thread(new Runnable() {
129             @Override
130             public void run() {
131                 service.listen();
132             }
133         });
134         serviceListenerThread.setDaemon(true);
135         serviceListenerThread.start();
136 
137         logger.logInfo("Embedded H2 database has started");
138     }
139 
140     /**
141      * Shutdown the in-memory H2 database by opening a connection and issuing
142      * the SHUTDOWN command.
143      *
144      * @param logger Used to report errors and raise exceptions.
145      */
146     public void stop(final Logger logger) {
147 
148         logger.logInfo("Stopping embedded H2 database");
149 
150         if (service != null) {
151             service.stop();
152         }
153 
154         logger.logInfo("Stopped embedded H2 database");
155     }
156 
157     @Override
158     public boolean isStarted(final Logger logger) {
159         return service != null && service.isRunning(true);
160     }
161 
162     @Override
163     public boolean isStopped(final Logger logger) {
164         return !service.isRunning(false);
165     }
166 }
