26  /**
27   * Abstract base classes for database objects providing an implementation of the
28   * {@link Database#load(com.btmatthews.utils.monitor.Logger, com.btmatthews.maven.plugins.inmemdb.Source)}
29   * operation that is used to load data into or execute scripts
30   * against the database.
31   *
32   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
33   * @version 1.0.0
34   */
35  public abstract class AbstractDatabase extends AbstractServer implements Database {
36  
37      /**
38       * The message key for the error reported when a server cannot be started.
39       */
40      protected static final String ERROR_STARTING_SERVER = "error_starting_server";
41      /**
42       * The message key for the error reported when a server cannot be stopped.
43       */
44      protected static final String ERROR_STOPPING_SERVER = "error_stopping_server";
45      /**
46       * The message key for the error reported when a file type is not supported.
47       */
48      private static final String UNSUPPORTED_FILE_TYPE = "unsupported_file_type";
49      private static final int PING_RETRIES = 10;
50      private static final int PING_DELAY = 500;
51      /**
52       * The database name.
53       */
54      private String databaseName;
55      /**
56       * The user name used to connect to the database.
57       */
58      private String databaseUsername;
59      /**
60       * The password used to connect to the database.
61       */
62      private String databasePassword;
63      /**
64       * The port number used for the database (not supported by all databases)
65       */
66      private int port;

68      /**
69       * Constructor initializes default database port.
70       *
71       * @param port The default port.
72       */
73      public AbstractDatabase(final int port) {
74          this.port = port;

77      /**
78       * Configure the server.
79       *
80       * @param name   Name of the configuration property.
81       * @param value  Value of the configuration property.
82       * @param logger Used for logging error and information messages.
83       */
84      @Override
85      public final void configure(final String name, final Object value, final Logger logger) {
86          if ("database".equals(name)) {
87              logger.logInfo("Configured database name: " + value);
88              databaseName = (String) value;
89          } else if ("username".equals(name)) {
90              logger.logInfo("Configured database username: " + value);
91              databaseUsername = (String) value;
92          } else if ("password".equals(name)) {
93              logger.logInfo("Configured database password: " + value);
94              databasePassword = (String) value;
95          } else if ("port".equals(name)) {
96              logger.logInfo("Configured database port: " + value);
97              port = (Integer) value;
98          }
99      }

146     protected abstract Loader[] getLoaders();

148     /**
149      * Find the loader that supports the source file and use it to load the data
150      * into or execute the script against the database.
151      *
152      * @param logger Used to report errors and raise exceptions.
153      * @param source The source file containing data or script.
154      */
155     @Override
156     public final void load(final Logger logger, final Source source) {
157         if (source == null) {
158             final String message = MessageUtil.getMessage(UNSUPPORTED_FILE_TYPE, "null");
159             logger.logError(message);
160         } else {
161             final Loader[] loaders = getLoaders();
162             int i = 0;
163             while (i < loaders.length) {
164                 if (loaders[i].isSupported(logger, source)) {
165                     loaders[i].load(logger, this, source);
166                     break;
167                 }
168                 ++i;
169             }
170             if (i >= loaders.length) {
171                 final String message = MessageUtil.getMessage(UNSUPPORTED_FILE_TYPE, source.getSourceFile());
172                 logger.logError(message);
173             }
174         }
175     }
176 
177     protected final boolean waitForStart() {
178         for (int i = 0; i < PING_RETRIES; i++) {
179             if (hasStarted()) {
180                 return true;
181             } else {
182                 try {
183                     Thread.sleep(PING_DELAY);
184                 } catch (final InterruptedException ie) {
185                     return false;
186                 }
187             }
188         }
189         return false;
190     }
191 
192     protected final boolean waitForStop() {
193         for (int i = 0; i < PING_RETRIES; i++) {
194             if (hasStopped()) {
195                 return true;
196             } else {
197                 try {
198                     Thread.sleep(PING_DELAY);
199                 } catch (final InterruptedException exception) {
200                     return false;
201                 }
202             }
203         }
204         return false;
205     }
