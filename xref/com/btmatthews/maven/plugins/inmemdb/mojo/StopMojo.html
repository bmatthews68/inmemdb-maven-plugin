17  package com.btmatthews.maven.plugins.inmemdb.mojo;
18  
19  import com.btmatthews.utils.monitor.mojo.AbstractStopMojo;
20  import org.apache.maven.plugin.MojoFailureException;
21  import org.apache.maven.plugins.annotations.LifecyclePhase;
22  import org.apache.maven.plugins.annotations.Mojo;
23  import org.apache.maven.plugins.annotations.Parameter;
24  
25  /**
26   * Stop an in-memory database server that is running as a daemon.
27   *
28   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
29   * @since 1.0.0
30   */
31  @Mojo(name = "stop", defaultPhase = LifecyclePhase.POST_INTEGRATION_TEST)
32  public class StopMojo extends AbstractStopMojo {
33  
34      /**
35       * Indicates whether or not the Mojo execution should be skipped.
36       *
37       * @since 1.4.0
38       */
39      @Parameter(property = "inmemdb.skip", defaultValue = "false")
40      private boolean skip;
41  
42      /**
43       * Checks whether the Mojo execution is being skipped before delegating to the super class to
44       * stop the in-memory database.
45       *
46       * @throws MojoFailureException If there was a problem executing the Mojo.
47       * @since 1.4.0
48       */
49      @Override
50      public void execute() throws MojoFailureException {
51          if (skip) {
52              getLog().info("Skipping inmemdb:stop because inmemdb.skip=='true'");
53          } else {
54              super.execute();
55          }
56      }
57  }
