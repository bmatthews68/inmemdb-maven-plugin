17  package com.btmatthews.maven.plugins.inmemdb.mojo;
18  
19  /**
20   * Describes source files that contain DBUnit data sets that can be used to
21   * populate the in-memory database.
22   *
23   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
24   * @version 1.0.0
25   * @see com.btmatthews.maven.plugins.inmemdb.Source
26   * @see AbstractSource
27   */
28  public final class DataSet extends AbstractSource {
29  
30      /**
31       * Indicates whether or not the data set contains qualified table names.
32       */
33      private Boolean qualifiedTableNames;
34  
35      /**
36       * The default constructor.
37       *
38       * @see AbstractSource#AbstractSource()
39       */
40      public DataSet() {
41      }
42  
43      /**
44       * Construct a source object that describes DBUnit data set.
45       *
46       * @param file The source file that contains the DDL/DML script or DBUnit
47       *             data set.
48       * @param flag <ul>
49       *             <li>{@link Boolean#TRUE} if the are qualified with the schema
50       *             name.</li>
51       *             <li>{@link Boolean#FALSE} if the are not qualified with the
52       *             schema name.</li>
53       *             </ul>
54       * @see AbstractSource#AbstractSource(String)
55       */
56      public DataSet(final String file, final Boolean flag) {
57          super(file);
58          this.qualifiedTableNames = flag;
59      }
60  
61      /**
62       * Determine if the DBUnit data set contains qualified or unqualified table
63       * names.
64       *
65       * @return <ul>
66       *         <li>{@link Boolean#TRUE} if the are qualified with the schema
67       *         name.</li>
68       *         <li>{@link Boolean#FALSE} if the are not qualified with the
69       *         schema name.</li>
70       *         </ul>
71       * @see com.btmatthews.maven.plugins.inmemdb.Source#getQualifiedTableNames()
72       */
73      public Boolean getQualifiedTableNames() {
74          return this.qualifiedTableNames;
75      }
76  
77      /**
78       * Indicate whether or not the DBUnit data set contains qualified or
79       * unqualified table names.
80       *
81       * @param flag <ul>
82       *             <li>{@link Boolean#TRUE} if the are qualified with the schema
83       *             name.</li>
84       *             <li>{@link Boolean#FALSE} if the are not qualified with the
85       *             schema name.</li>
86       *             </ul>
87       */
88      public void setQualifiedTableNames(final Boolean flag) {
89          this.qualifiedTableNames = flag;
90      }
91  
92      @Override
93      public String toString() {
94          return "DataSet[" + getSourceFile() + "]";
95      }
96  }
