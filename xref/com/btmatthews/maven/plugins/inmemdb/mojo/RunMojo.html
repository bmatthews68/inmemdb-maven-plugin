33  /**
34   * This plug-in Mojo starts an In Memory Database.
35   *
36   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
37   * @version 1.2.0
38   */
39  @Mojo(name = "run", defaultPhase = LifecyclePhase.PRE_INTEGRATION_TEST)
40  public final class RunMojo extends AbstractRunMojo {
41  
42      /**
43       * The source files used to populate the database.
44       */
45      @Parameter
46      private List<? extends Source> sources;
47      /**
48       * The database type.
49       */
50      @Parameter(property = "inmemdb.type", defaultValue = "hsqldb")
51      private String type = "hsqldb";
52      /**
53       * The database name.
54       */
55      @Parameter(property = "inmemdb.database", defaultValue = ".")
56      private String database = ".";
57      /**
58       * The username for database connections.
59       */
60      @Parameter(property = "inmemdb.username", defaultValue = "sa")
61      private String username = "sa";
62      /**
63       * The password for database connections.
64       */
65      @Parameter(property = "inmemdb.password", defaultValue = "")
66      private String password = "";
67      /**
68       * The port for database connections.
69       */
70      @Parameter(property = "inmemdb.port")
71      private Integer port;
72  
73      /**
74       * Additional attributes used to configure the database connection.
75       */
76      @Parameter(property = "inmemdb.attribute")
77      private Map<String, String> attributes;
78  
79      /**
80       * Indicates whether or not the Mojo execution should be skipped.
81       * @since 1.4.0
82       */
83      @Parameter(property = "inmemdb.skip", defaultValue = "false")
84      private boolean skip;
85  
86      /**
87       * Checks whether the Mojo execution is being skipped before delegating to the super class to
88       * run the in-memory database.
89       *
90       * @throws MojoFailureException If there was a problem executing the Mojo.
91       * @since 1.4.0
92       */
93      @Override
94      public void execute() throws MojoFailureException {
95          if (skip) {
96              getLog().info("Skipping inmemdb:run because inmemdb.skip=='true'");
97          } else {
98              super.execute();
99          }
100     }
101 
102     /**
103      * Get the server type.
104      *
105      * @return The value of {@link #type}.
106      */
107     @Override
108     public String getServerType() {
109         return type;
110     }
111 
112     /**
113      * Get the server configuration parameters. These are {@link #database}, {@link #port}, {@link #username} and {@link #password}.
114      *
115      * @return A {@link Map} containing the configuration parameters.
116      */
117     @Override
118     public Map<String, Object> getServerConfig() {
119         final Map<String, Object> config = new HashMap<String, Object>();
120         config.put("database", database);
121         config.put("username", username);
122         if (port != null) {
123             config.put("port", port);
124         }
125         if (password == null) {
126             config.put("password", "");
127         } else {
128             config.put("password", password);
129         }
130 
131         if (attributes != null) {
132             config.put("attributes", attributes);
133         }
134 
135         return config;
136     }
137 
138     /**
139      * This callback is invoked after the server has started and is used load the scripts
140      * and datasets that will initialise the database.
141      *
142      * @param server The database server.
143      * @param logger Used to log information and error messages.
144      */
145     @Override
146     public void started(final Server server, final Logger logger) {
147         logger.logInfo("Server has been started");
148         if (sources != null) {
149             logger.logInfo("Executing initialization scripts and loading data sets");
150             for (final Source source : sources) {
151                 logger.logInfo("Loading " + source.toString());
152                 ((Database) server).load(this, source);
153             }
154         }
155     }
156 }
